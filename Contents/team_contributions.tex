\begin{table}[H]
	\centering
	\begin{tabular}{|p{1cm}|p{3.5cm}|p{8.5cm}|p{2cm}|}
		\hline
		\textbf{Task} & \textbf{Họ tên} & \textbf{Nội dung thực hiện} & \textbf{Đóng góp} \\
		\hline
		
		\textbf{\#1} &
		Mai Bách Huyên &
		\textbf{Giao diện:}
		\begin{itemize}
			\item Trang chủ.
			\item Trang liên hệ.
		\end{itemize}
		
		\textbf{Tính năng quản lý dành cho quản trị viên:}
		\begin{itemize}
			\item Quản lý thông tin trên các trang đã thiết kế như nội dung giới thiệu, số điện thoại, địa chỉ, hình ảnh, logo.
			\item Quản lý các liên hệ của khách hàng: xem thông tin, đánh dấu đã đọc/chưa đọc/đã phản hồi, xoá liên hệ.
		\end{itemize}
		&
		100\% \\
		\hline
		
		\textbf{\#3} &
		Nguyễn Duy Nhất &
		\textbf{Giao diện:}
		\begin{itemize}
			\item Trang danh sách sản phẩm, cho phép tìm kiếm sản phẩm theo từ khoá.
			\item Trang thông tin chi tiết sản phẩm và giỏ hàng.
		\end{itemize}
		
		\textbf{Tính năng quản lý dành cho quản trị viên:}
		\begin{itemize}
			\item Quản lý sản phẩm: xem, tìm kiếm, thêm, sửa, xoá thông tin sản phẩm.
			\item Quản lý giỏ hàng, đơn hàng: xem thông tin, đánh dấu trạng thái cho giỏ hàng và đơn hàng.
		\end{itemize}
		&
		100\% \\
		\hline
		
		\textbf{\#4} &
		Lữ Hoàng Duy &
		\textbf{Giao diện:}
		\begin{itemize}
			\item Trang danh sách bài viết, cho phép tìm kiếm bài viết hoặc tin tức theo từ khoá.
			\item Trang đọc bài viết.
		\end{itemize}
		
		\textbf{Tính năng quản lý dành cho quản trị viên:}
		\begin{itemize}
			\item Quản lý tin tức: xem, tìm kiếm, thêm, sửa, xoá thông tin bài viết.
			\item Quản lý bình luận/đánh giá của người dùng trên bài viết.
		\end{itemize}
		&
		100\% \\
		\hline
